43  XForm(const CGLA::Vec3d& _llf, const CGLA::Vec3d& _urt, const CGLA::Vec3i& _DIM, double _margin=0.1):
44  llf(_llf), urt(_urt), DIM(_DIM)
45  {
46  if(urt[0]<llf[0]) {
47  scale = 1;
48  offset = CGLA::Vec3d(0.0);
49  llf = CGLA::Vec3d(0);
50  urt = CGLA::Vec3d(DIM-CGLA::Vec3i(1));
51  }
52  else {
53  CGLA::Vec3d d = urt - llf;
54  double margin = d.max_coord() * _margin;
55  double scale_x = (DIM[0]-1)/(d[0] + 2.0 * margin);
56  double scale_y = (DIM[1]-1)/(d[1] + 2.0 * margin);
57  double scale_z = (DIM[2]-1)/(d[2] + 2.0 * margin);
58  scale = fmin(fmin(scale_x, scale_y), scale_z);
59  offset = CGLA::Vec3d(margin);
60  }
61  }
